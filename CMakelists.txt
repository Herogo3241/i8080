cmake_minimum_required(VERSION 3.20)
project(i8080)

set(CMAKE_CXX_STANDARD, 23)


set(IMGUI_DIR vendored/imgui)
set(IMGUI_SOURCES
    ${IMGUI_DIR}/imgui.cpp
    ${IMGUI_DIR}/imgui_draw.cpp
    ${IMGUI_DIR}/imgui_tables.cpp
    ${IMGUI_DIR}/imgui_widgets.cpp
    ${IMGUI_DIR}/backends/imgui_impl_glfw.cpp
    ${IMGUI_DIR}/backends/imgui_impl_opengl3.cpp
)

set(GLAD_DIR vendored/glad)



file(GLOB_RECURSE SOURCES "src/*.cpp")



set(CMAKE_RUNTIME_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/$<CONFIGURATION>")
set(CMAKE_LIBRARY_OUTPUT_DIRECTORY "${CMAKE_BINARY_DIR}/$<CONFIGURATION>")

add_subdirectory(vendored/glfw3)

add_library(glad ${GLAD_DIR}/src/glad.c)
target_include_directories(glad PUBLIC ${GLAD_DIR}/include)



add_executable(i8080 WIN32 ${SOURCES} ${IMGUI_SOURCES})

target_include_directories(i8080 PRIVATE ${IMGUI_DIR} vendored/glfw3/include)
target_link_libraries(i8080 PRIVATE
    glfw
    glad
    opengl32   
)
